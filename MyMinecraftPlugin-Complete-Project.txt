import com.yourname.myplugin.blocks.CustomBlockListener
import com.yourname.myplugin.blocks.CustomBlocks
import com.yourname.myplugin.commands.BlocksCommand
import com.yourname.myplugin.commands.ExampleCommand
import com.yourname.myplugin.commands.ReloadCommand
import com.yourname.myplugin.graphics.TextureCommand
import com.yourname.myplugin.graphics.TextureManager
import com.yourname.myplugin.listeners.ExampleListener
import com.yourname.myplugin.utils.ConfigManager
import kotlinx.coroutines.*
import net.kyori.adventure.text.Component
import net.kyori.adventure.text.format.NamedTextColor
import org.bukkit.plugin.java.JavaPlugin
import kotlin.coroutines.CoroutineContext

class MyPlugin : JavaPlugin(), CoroutineScope {
    
    private val job = SupervisorJob()
    override val coroutineContext: CoroutineContext
        get() = Dispatchers.Main + job
    
    lateinit var configManager: ConfigManager
        private set
    lateinit var customBlocks: CustomBlocks
        private set
    lateinit var textureManager: TextureManager
        private set
        
    val customBlockData = mutableMapOf<String, String>()
    
    override fun onEnable() {
        configManager = ConfigManager(this)
        configManager.loadConfig()
        
        textureManager = TextureManager(this)
        customBlocks = CustomBlocks(this)
        customBlocks.registerCustomBlocks()
        
        if (configManager.customTexturesEnabled) {
            launch {
                delay(2000)
                textureManager.initialize()
            }
        }
        
        getCommand("example")?.setExecutor(ExampleCommand(this))
        getCommand("reload")?.setExecutor(ReloadCommand(this))
        getCommand("blocks")?.setExecutor(BlocksCommand(this))
        getCommand("textures")?.setExecutor(TextureCommand(this))
        
        server.pluginManager.registerEvents(ExampleListener(this), this)
        server.pluginManager.registerEvents(CustomBlockListener(this), this)
        
        launch {
            delay(1000)
            logger.info("Async startup completed!")
        }
        
        logger.info("MyMinecraftPlugin v${description.version} has been enabled!")
        server.consoleSender.sendMessage(
            Component.text("Plugin loaded successfully! Built with Gradle + Kotlin")
                .color(NamedTextColor.GREEN)
        )
        
        if (configManager.customBlocksEnabled) {
            server.consoleSender.sendMessage(
                Component.text("Custom blocks system enabled!")
                    .color(NamedTextColor.AQUA)
            )
        }
        
        if (configManager.customTexturesEnabled) {
            server.consoleSender.sendMessage(
                Component.text("Custom textures system enabled!")
                    .color(NamedTextColor.LIGHT_PURPLE)
            )
        }
    }
    
    override fun onDisable() {
        job.cancel()
        logger.info("MyMinecraftPlugin has been disabled!")
        
        server.consoleSender.sendMessage(
            Component.text("Plugin unloaded successfully!")
                .color(NamedTextColor.YELLOW)
        )
    }
    
    fun reload() {
        launch {
            configManager.reloadConfig()
            
            if (configManager.customBlocksEnabled) {
                customBlocks.registerCustomBlocks()
            }
            
            if (configManager.customTexturesEnabled) {
                textureManager.regenerateResourcePack()
            }
            
            logger.info("Plugin configuration reloaded!")
            
            server.consoleSender.sendMessage(
                Component.text("Configuration reloaded!")
                    .color(NamedTextColor.GREEN)
            )
        }
    }
    
    fun runAsync(block: suspend CoroutineScope.() -> Unit) {
        launch(Dispatchers.IO) {
            block()
        }
    }
}\n\n============================================================\n\nüìÑ FILE: README.md\n-----------------\n# MyMinecraftPlugin (Kotlin + Paper + Gradle)

A modern Minecraft plugin for Paper servers, built with Kotlin and Gradle.

## Features

- **Modern Kotlin**: Written entirely in Kotlin with coroutines for async operations
- **Paper API**: Uses Paper's enhanced API with Adventure text components  
- **Gradle Build**: Modern build system with Kotlin DSL
- **Custom Blocks**: Magic Stone, Enchanted Log, Crystal Block
- **Custom Textures**: Automatic resource pack generation and distribution

## Commands

- `/example` - Basic greeting command with async processing
- `/blocks give <type> [player] [amount]` - Give custom blocks
- `/textures download` - Get custom textures resource pack

## Installation

1. Ensure you're running Paper 1.20.4 or later
2. Place the plugin JAR in your server's `plugins/` folder
3. Restart your server
4. Configure in `plugins/MyMinecraftPlugin/config.yml`

## Building

```bash
./gradlew build          # Build the plugin
./gradlew runServer      # Start test server
```
